var num$
var resultado$
var zero$
var num2$
var dois$
var j$
var cont$

zero$ = 0;
num2$ = 1;
dois$ = 2;

IMPRIMA('Programa para testar se um numero eh par ou impar e primo ou nao');

IMPRIMA('Digite o seu numero');

ESCREVA(num$);

resultado$ = num$%2;

SE(resultado$@zero$)
	IMPRIMA('Eh par');
FIM;

SE(resultado$#zero$)
	IMPRIMA('Eh impar');
FIM;

j$=num$+1;
LACO(num2$<j$)
	resultado$=num$%num2$;
	SE(resultado$@zero$)
		cont$=cont$+1;
	FIM;
	num2$=num2$+1;
FIMLACO;

SE(cont$@dois$)
	IMPRIMA('Eh primo');
FIM;

SE(cont$#dois$)
	IMPRIMA('Nao eh primo');
FIM;			

IMPRIMA('Programa para calcular a media de dois numeros');

IMPRIMA('Digite o primeiro numero');

ESCREVA(num$);

IMPRIMA('Digite o segundo numero');

ESCREVA(num2$);

cont$=num$+num2$;

resultado$=cont$/2;

IMPRIMA(resultado$);
